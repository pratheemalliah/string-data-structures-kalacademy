{
    function isSubSeq(str1,str2) {
        if(str1.length == 0) return true;

        let i=0,j=0;
        while(i<str1.length && j<str2.length) {
            if(str1[i] == str2[j]) {
                i++;
            }
            if(str1.length ==  i) {
                return true;
            }
            j++;
        }
        return false;
    }

    isSubSeq('ade','abckefgh')
}
